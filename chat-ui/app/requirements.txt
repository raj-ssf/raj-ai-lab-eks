# Chainlit pinned exactly to 2.11.1 (was: chainlit>=2.0.0,<3.0.0).
#
# Pin tightened after the Phase #45+#47 canary regression: a fresh
# image build with the range pin resolved a different chainlit
# minor version, which serialized session cookies in a format
# subtly incompatible with the prior version. Existing browsers
# carrying cookies issued by the OLD version's chainlit hit the
# NEW version's deserializer and ended up with empty
# user.metadata["access_token"] → chat-ui forwarded an empty
# Bearer token to langgraph-service → langgraph returned 401.
# All caught by the canary at 25% traffic share before going
# 100%. Phase #45 paid for itself.
#
# Bumping chainlit is now an explicit decision, not pip's choice.
# When ready to bump: change the pin, rebuild, push to canary,
# observe.
chainlit==2.11.1

# Async HTTP client to call langgraph-service /invoke. Pinned to match
# the version in langgraph-service/rag-service for consistent behavior
# (timeout, mTLS quirks, etc. are sensitive to httpx version).
httpx==0.27.2

prometheus-client==0.21.0
