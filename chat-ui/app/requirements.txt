# Chainlit 1.5+ — earlier versions (1.3.x) hardcoded their built-in
# OAuth provider list (Google/GitHub/Azure-AD/Okta/Auth0/...) without
# Keycloak. Appending a custom KeycloakOAuthProvider at runtime
# satisfied the decorator's import-time check but didn't register the
# /auth/oauth/keycloak/{login,callback} HTTP routes (route
# registration in 1.3 enumerates the hardcoded list, not the runtime
# providers list). Result: browser hit /auth/oauth/keycloak/login,
# got the SPA HTML instead of a redirect to Keycloak, looped.
# 1.5+ has Keycloak in the built-in list and registers its routes
# correctly from the OAUTH_KEYCLOAK_* env vars.
chainlit>=1.5.0,<2.0.0
